# See traditions/README.txt for format
# These overrides are here to make it less likely you accidentally take both the AP and Terraform traditions (Detox/World Shapers)

#################
# GENERIC ASCENSION PERKS
#################

ap_world_shaper = {
	possible = {
		custom_tooltip = {
			fail_text = "requires_technology_tech_climate_restoration"
			has_technology = tech_climate_restoration
		}
		custom_tooltip = {
			fail_text = "requires_ascension_perks_1"
			num_ascension_perks > 0
		}
		custom_tooltip = {
			fail_text = "tt_already_has_terraform_tradition"
			NOT = { has_tradition = "tr_terraform_adopt" }
		}
	}

	modifier = {
		terraforming_cost_mult = -0.25
		add_attunement_the_composer_of_strands = @ascension_perk_attunement1
	}

	on_enabled = {
		custom_tooltip = "allow_gaia"
	}

	potential = {
		NOT = {
			has_ascension_perk = ap_world_shaper
		}
		OR = {
			is_hive_empire = no
			is_wilderness_empire = yes
		}
		OR = {
			is_machine_empire = no
			has_valid_civic = civic_machine_servitor
			has_valid_civic = civic_machine_assimilator
		}
	}

	ai_weight = {
		factor = 5
		modifier = {
			factor = 0
			OR = {
				is_void_dweller_empire = yes
				has_origin = origin_shattered_ring
				has_trait = trait_aquatic
				has_ethic = ethic_gestalt_consciousness
				has_tradition = tr_terraform_adopt
			}
		}
		modifier = {
			factor = 2
			OR = {
				has_ethic = ethic_pacifist
				has_ethic = ethic_xenophobe
			}
		}
		modifier = {
			factor = 3
			OR = {
				has_ethic = ethic_fanatic_pacifist
				has_ethic = ethic_fanatic_xenophobe
				is_wilderness_empire = yes
			}
		}
		modifier = { #only way to get more nice planets
			factor = 100
			has_origin = origin_life_seeded
		}
	}
}

#################
# TOXIC ASCENSION PERKS
#################

ap_detox = {
	potential = {
		host_has_dlc = "Toxoids Species Pack"
		NOT = {
			has_ascension_perk = ap_detox
			has_country_flag = knights_detox
		}
	}

	possible = {
		custom_tooltip = {
			fail_text = "requires_technology_tech_climate_restoration"
			has_technology = tech_climate_restoration
		}
		custom_tooltip = {
			fail_text = "tt_already_has_terraform_tradition"
			NOT = { has_tradition = "tr_terraform_adopt" }
		}
	}

	on_enabled = {
		custom_tooltip = "detox_tox_worlds"
	}

	modifier = {
		add_attunement_the_composer_of_strands = @ascension_perk_attunement1
	}

	ai_weight = {
		factor = 20
		modifier = {
			factor = 0
			OR = {
				has_tradition = tr_terraform_adopt
				NOT = {
					any_planet_within_border = {
						has_modifier = toxic_terraforming_candidate
					}
				}
			}
		}
		modifier = {
			factor = 3
			count_planet_within_border = {
				count > 2
				limit = { has_modifier = toxic_terraforming_candidate }
			}
		}
	}
}

ap_tt_creator_of_worlds = {
	potential = {
		NOT = {
			has_ascension_perk = ap_tt_creator_of_worlds
		}
	}

	possible = {
		custom_tooltip = {
			fail_text = "requires_technology_tech_climate_restoration"
			has_technology = tech_climate_restoration
		}
		custom_tooltip = {
			fail_text = "requires_ascension_perks_3"
			num_ascension_perks > 2
		}
		custom_tooltip = {
			fail_text = "tt_requires_terraform_tradition_finish"
			has_tradition = tr_terraform_finish
		}
	}

	on_enabled = {
		custom_tooltip = "tt_creator_of_worlds_unlock_tooltip"
		hidden_effect = {
			add_modifier = {
				modifier = tt_creator_of_worlds_gaia_influence_modifier
				multiplier = value:tt_creator_of_worlds_gaia_influence
			}
		}
	}

	# Player only
	ai_weight = {
		factor = 0
	}
}
