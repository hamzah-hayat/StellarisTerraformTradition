# Terraform Tradition effects/perks

# Adopt
# Unlock Terraforming Agenda and gives basic terraforming tech
# Also makes clearing blockers faster and cheaper
tr_terraform_adopt = {
	unlocks_agenda = agenda_new_worlds
	modifier = {
		planet_clear_blocker_speed_mult = 0.33
		deposit_blockers_cost_mult = -0.5
	}
	
	on_enabled = {
		if = {
			limit = { NOT = { has_technology = tech_terrestrial_sculpting } }
			add_research_option = tech_terrestrial_sculpting
			add_tech_progress = {
				tech = tech_terrestrial_sculpting
				progress = 0.25
			}
		}

		if = {
			limit = {
				has_ascension_perk = ap_world_shaper
			}
			remove_ascension_perk = ap_world_shaper
		}

		if = {
			limit = {
				has_ascension_perk = ap_detox
			}
			remove_ascension_perk = ap_detox
		}
	}
}

# Finish
# Gives you a special Gaia Mastery planet preference type, and an ascension perk point
# Also unlocks the Creator of Worlds ascension perk
tr_terraform_finish = {
	modifier = {
		ascension_perks_add = 1
	}

	on_enabled = {
		custom_tooltip = "tr_terraform_finish_gaia_mastery_unlocked"
		custom_tooltip = "tr_terraform_finish_creator_of_worlds_unlocked"
	}

	ai_weight = {
		factor = 5000
	}
}

# Frontier Exploitation
# Increases Job efficiency of Rural jobs (Farmers, Miners, Technicians)
# Also gives a bonus to food/minerals/energy produced by those jobs
tr_terraform_frontier_exploitation = {
	modifier = {
		planet_miners_produces_mult = 0.15
		planet_farmers_produces_mult = 0.15
		planet_technician_produces_mult = 0.15
	}

	ai_weight = {
		factor = 5000
	}
}

# Planet Probing
# Enables decisions to add a single planetary deposit to a planet
# Plus a bonus to rare crystal/exotic gas/volatile mote production
tr_terraform_planet_probing = {
	modifier = {
		planet_jobs_exotic_gases_produces_mult = 0.1
		planet_jobs_rare_crystals_produces_mult = 0.1
		planet_jobs_volatile_motes_produces_mult = 0.1
	}

	on_enabled = {
		custom_tooltip = "tr_terraform_create_basic_planetary_deposit"
		custom_tooltip = "tr_terraform_create_strategic_planetary_deposit"
	}

	ai_weight = {
		factor = 1000
	}
}

# World Shaper
# Same effect as the World Shaper ascension perk
tr_terraform_world_shaper = {
	possible = {
		has_tradition = tr_terraform_frontier_exploitation
		has_tradition = tr_terraform_planet_probing
	}

	modifier = {
		terraforming_cost_mult = -0.25
	}

	on_enabled = {
		custom_tooltip = "allow_gaia"

		if = {
			limit = { NOT = { has_technology = tech_ecological_adaptation } }
			add_research_option = tech_ecological_adaptation
			add_tech_progress = {
				tech = tech_ecological_adaptation
				progress = 0.25
			}
		}
	}

	ai_weight = {
		factor = 1000
	}
}

# Climate Engineering
# Reduces terraforming time and cost
tr_terraform_climate_engineering = {
	modifier = {
		terraforming_cost_mult = -0.25
		terraform_speed_mult = 0.25
	}

	ai_weight = {
		factor = 1000
	}
}

# Planet Restoration
# Lets you Terraform all terraforming candidate planets, eg Barren, Toxic and Frozen
tr_terraform_planet_restoration = {
	possible = {
		has_tradition = tr_terraform_climate_engineering
	}

	on_enabled = {
		custom_tooltip = "tr_terraform_planet_restoration_unlocked"
		if = {
			limit = { NOT = { has_technology = tech_climate_restoration } }
			add_research_option = tech_climate_restoration
			add_tech_progress = {
				tech = tech_climate_restoration
				progress = 0.25
			}
		}
	}

	ai_weight = {
		factor = 1000
	}
}
