# Decisions for the Terraform Tradition

## Planet Probing Decisions
## Basic Deposits
### Adds a basic planetary deposit to a planet, with the type of deposit depending on the planet class. 
tt_decision_add_planetary_deposit_energy = {
	icon = decision_divine_algorithm
	sound = decision_resources

	enactment_time = 360

	resources = {
		category = decisions
		cost = { energy = 1500 }
	}

	potential = {
		exists = owner
		owner = {
			has_tradition = tr_terraform_planet_probing
		}
		NOT = { has_planet_flag = tt_established_basic_deposit }
		habitable_planet_not_urban = yes
	}

	owned_planets_only = yes

	on_queued = {
		planet = {
			set_planet_flag = tt_established_basic_deposit
		}
	}

	on_unqueued = {
		planet = {
			remove_planet_flag = tt_established_basic_deposit
		}
	}


	effect = {
		planet = {
			if = {
				limit = { 
					OR = {
						is_planet_class = pc_arctic
						is_planet_class = pc_tundra
						is_planet_class = pc_savannah
						is_planet_class = pc_desert
						is_planet_class = pc_arid
						is_planet_class = pc_volcanic
					}
				}
				add_deposit = d_geothermal_vent
			}
			else_if = {
				limit = { 
					OR = {
						is_planet_class = pc_gaia
						is_planet_class = pc_relic
						is_planet_class = pc_ocean
						is_planet_class = pc_nuked
					}
				}
				add_deposit = d_underwater_vent
			}
			else = {
				add_deposit = d_tempestous_mountain
			}
			custom_tooltip = tt_decisions_add_basic_deposit_block_tooltip
		}
	}

	ai_weight = {
		weight = 1
		modifier = {
			factor = 10
			OR = {
				has_designation = col_generator
				AND = {
					has_district = district_generator
					num_free_districts = { type = district_generator value < 3 }
				}
			}
		}
	}
}

## Strategic Deposits
### Adds a rare crystals deposit to a planet
### Note, Volcanic worlds do not have a rare crystals deposit
tt_decision_add_planetary_deposit_rare_crystals = {
	icon = decision_add_planetary_deposit_rare_crystals
	sound = decision_resources

	enactment_time = 360

	resources = {
		category = decisions
		cost = { energy = 3000 }
	}

	potential = {
		exists = owner
		owner = {
			has_tradition = tr_terraform_planet_probing
		}
		NOT = { is_planet_class = pc_volcanic }
		NOT = { has_planet_flag = tt_established_strategic_deposit }
		habitable_planet_not_urban = yes
	}

	allow = {
		owner = { 
			OR = {
				has_technology = tech_mine_rare_crystals
				has_technology = tech_rare_crystals
			}
		}
	}

	owned_planets_only = yes

	on_queued = {
		planet = {
			set_planet_flag = tt_established_strategic_deposit
		}
	}

	on_unqueued = {
		planet = {
			remove_planet_flag = tt_established_strategic_deposit
		}
	}

	effect = {
		planet = {
			if = {
				limit = {
					OR = {
						is_planet_class = pc_ocean
						is_planet_class = pc_nuked
						is_planet_class = pc_gaia
						is_planet_class = pc_hive
						is_planet_class = pc_machine
					}
				}
				add_deposit = d_crystal_reef
			}
			else = {
				add_deposit = d_crystal_forest
			}
			custom_tooltip = tt_decisions_add_strategic_deposit_block_tooltip
		}
	}

	ai_weight = {
		weight = 1
		modifier = {
			factor = 10
			OR = {
				has_designation = col_mining
				owner = {
					has_monthly_income = { resource = rare_crystals value <= 1 }
				}
			}
		}
	}
}

## Bonus decision
### Allows you to replace Mastery of Nature with Worldscaping if you have the civic
decision_cultivated_worldscaping_replace_mastery_of_nature = {
	owned_planets_only = yes
	sound = event_administrative_work
	icon = decision_cultivated_worldscaping

	resources = {
		category = decisions
		cost = {
			energy = 2000
			influence = 100
		}
	}

	enactment_time = 360

	potential = {
		exists = owner
		owner = {
			is_environmental_architects_empire = yes
		}
		habitable_planet_not_urban = yes
		has_modifier = mastery_of_nature
	}

	allow = {
		hidden_trigger = {
			check_variable_arithmetic = {
				which = value:natural_blocker_count
				value = 0
			}
		}
		custom_tooltip = {
			fail_text = "decision_cultivated_worldscaping_natural_blockers_fail_tt"
			check_variable_arithmetic = {
				which = value:natural_blocker_count
				value = 0
			}
		}
	}

	effect = {
		remove_modifier = mastery_of_nature
		add_modifier = {
			modifier = cultivated_worldscaping_modifier
		}
	}

	# You would always want to replace Mastery of Nature with Worldscaping, so AI should as well!
	ai_weight = {
		weight = 100
	}
}
